int total = 0
calculate total after each iteration:
	add mark to total

calculate mean:
	if the counter is > 0:
		mean = total/counter
	else
		mean = 0

calulate variance:
	double variance = 0
	loop through the marks:
		for each mark, substract the mean and square the result
		add the result to variance

calculate sample variance:
	divide variance by counter -1

calculate standard deviation:
	get the square root of variance

print mean
print standard deviation
