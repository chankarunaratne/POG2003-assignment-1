create array “marks” containing 30 elements
set maxMark to -1
set minMark to 101
set counter = 1

do the following as long as the counter is <=30:
	ask user for mark
	add mark into the “marks” array at the position [counter -1]
	if mark > maxMark then
		set maxMark to mark
	if mark < minMark then
		set minMark to mark
	increase counter by 1

print “The highest mark is ” maxMark
print “The lowest mark is ” minMark
